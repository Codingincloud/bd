\section{Introduction}

\begin{frame}{Project Overview}
    \begin{block}{Blood Donor Information Management System}
        A comprehensive web-based application designed to centralize and manage detailed blood donor information for healthcare institutions.
    \end{block}
    
    \begin{itemize}
        \item Built with Django Framework and PostgreSQL
        \item Focuses on comprehensive donor profile management
        \item Provides health metrics tracking and monitoring
        \item Enables advanced search and filtering capabilities
        \item Implements automated eligibility calculations
    \end{itemize}
\end{frame}

\begin{frame}{Problem Statement}
    \begin{alertblock}{Current Challenges}
        Healthcare institutions face significant challenges:
    \end{alertblock}
    
    \begin{itemize}
        \item Scattered medical records and donor information
        \item Manual health metrics tracking processes
        \item Inefficient donor search systems
        \item Lack of centralized donor databases
        \item Complex blood compatibility tracking
        \item Manual eligibility calculations
        \item Poor emergency response coordination
    \end{itemize}
\end{frame}

\begin{frame}{Objectives}
    \begin{block}{Primary Objective}
        Create a centralized donor information management platform that organizes, stores, and manages comprehensive donor data efficiently.

    \textbf{Specific Goals:}
    \begin{itemize}
        \item Develop detailed donor profile management
        \item Implement health metrics tracking system
        \item Create advanced search and filtering capabilities
        \item Establish automated eligibility calculations
        \item Provide comprehensive donor analytics
        \item Enable emergency request handling
    \end{itemize}
\end{frame}

\section{System Workflow}

\begin{frame}{System Algorithm}
    \begin{algorithm}[H]
        \caption{Blood Donor Management System Workflow}
        \begin{algorithmic}[1]
            \STATE Start
            \STATE Ask user: "Do you have an account?"
            \IF{No}
                \STATE Create new account (Donor/Admin registration)
            \ENDIF
            \STATE Enter login credentials
            \IF{Invalid credentials}
                \STATE Prompt to re-enter credentials
            \ELSE
                \STATE Display appropriate dashboard (Donor/Admin)
            \ENDIF
            \STATE Access system features based on user role
            \IF{Donor}
                \STATE View profile, donation history, health metrics
                \STATE Update personal information
                \STATE Check eligibility status
            \ELSE
                \STATE Manage donor database
                \STATE Handle donation requests
                \STATE Generate reports and analytics
            \ENDIF
            \STATE Stop
        \end{algorithmic}
    \end{algorithm}
\end{frame}
